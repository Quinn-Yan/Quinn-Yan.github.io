小程序--使用缓存实现文章收藏

Quinnyan
2018-11-10

Page({

data:{

},

onLoad: function(option) {

var postId = option.id;

this.data.currentPostId = postId; 

var postData = postsData.postList[postId];

this.setData({

  //如果在onLoad方法中，不是异步的去执行一个数据绑定

  //则不需要使用this.setData方法

  //只需要使用this.data赋值即可实现数据绑定

  //凡是涉及到数据绑定，用this.setData就行了。不管是不是在onLoad方法里

  postData: postData

})

var postsCollected = wx.getStorageSync('posts_collected') 

//获取所有数据缓存

if (postsCollected //判断此数据缓存是否存在) {

  var postCollected = postsCollected[postId];

//存在则执行这段代码获取所有数据缓存的其中一条 

if (postCollected //判断是false/true) {

   this.setData({

  collected: postCollected

//将postCollected是false/true绑定到collected，而collected又绑定传值在wx:if=”“中； 

  })

  }

}

else{

//一开始执行onLoad方法中postsCollected是没有的，所以判断中是执行了else的里的代码

  var postsCollected = {}; //一开始缓存数据为空，

  postsCollected[postId] = false; //所以所有的页面内容都没有被收藏

  wx.setStorageSync('posts_collected', postsCollected);

  //设置一个缓存值

}

},

onColletionTap: function(event) {

var postsCollected = wx.getStorageSync('posts_collected')

var postCollected = postsCollected[this.data.currentPostId];

// 收藏变成未收藏，未收藏变成收藏

postCollected = !postCollected; //取反，false变成true；

postsCollected[this.data.currentPostId] = postCollected;

// 更新文章是否的缓存值

wx.setStorageSync('posts_collected', postsCollected);

// 更新数据绑定变量，从而实现切换图片，就是未收藏变成收藏的缓存值设置进去缓存值里

  this.setData ({

  collected: postCollected //取反的值（false变成了true）传递到wx:if="collected"来决定下次点击，再取反由true变成false；

})

}

})

当点击页面进入的时候先执行onLoad会把里面的代码都跑完，postsCollected还没设置数据和缓存值，也就是执行了else里的代码，所以所有的postId都是false；当点击一次触摸收藏后就有第一次数据并且有了缓存值；下次点开页面先执行完onLoad里面的代码也就有缓存值和数据了。就会显示上次点击收藏的文章数据是true即显示已收藏的图标；
